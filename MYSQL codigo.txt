CREATE DATABASE TextilesHogar;
USE TextilesHogar;
CREATE TABLE clientes (
  idcliente INT AUTO_INCREMENT PRIMARY KEY,
  nombres   VARCHAR(100) NOT NULL,
  textil    VARCHAR(100) NOT NULL,     
  stock     INT NOT NULL,
  precio    DECIMAL(10,2) NOT NULL,
  genero    VARCHAR(20),                 
  direccion VARCHAR(200),
  telefono  VARCHAR(25),
  email     VARCHAR(120)
);
CREATE TABLE login (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50),
    password VARCHAR(50)
);
